def sheffe(c, samples):
    ns = np.array([len(x) for x in samples])
    k = len(samples)
    n = np.array(ns).sum()
    sin = 1. / (n - k) * (ns * np.array([x.var() for x in samples])).sum()
    S = (c * np.array([x.mean() for x in samples])).sum() ** 2 / ((k - 1) * sin * (c ** 2 / ns).sum())
    return S
